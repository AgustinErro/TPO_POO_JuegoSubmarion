@startuml
title R1 - Navegación del Submarino

actor "jugador:Jugador" as jugador
participant "controlador:Controlador" as controlador
participant "juego:JuegoSubmarino" as juego
participant "sub:Submarino" as sub
participant "area:Area" as area

jugador -> controlador : 1: moverDerechaSubmarino()
activate controlador

controlador -> juego : 2: terminarJuego()
activate juego
juego -> sub : 3: getVidas()
activate sub
sub --> juego : vidas
deactivate sub
juego --> controlador : false
deactivate juego

controlador -> juego : 4: moverDerecha()
activate juego
juego -> sub : 5: moverDerecha()
activate sub
sub -> area : 6: estaDentroHorizontal(nuevaX, ancho)
activate area
area --> sub : true / false
deactivate area

opt [dentro del área]
    sub -> sub : 7: posX = nuevaX
end

sub --> juego : posX
deactivate sub
deactivate juego
deactivate controlador
@enduml
