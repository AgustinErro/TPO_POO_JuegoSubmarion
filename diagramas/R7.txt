@startuml
title R7 - Gestión de Vidas

participant "juego:JuegoSubmarino" as juego
participant "sub:Submarino" as sub

== Perder una vida ==

sub -> sub : 1: calcularDano(distancia)
activate sub
sub -> sub : 2: perderVida()
sub -> sub : 3: vidas--
alt [vidas vidas >= 0]
    sub -> sub : 4a: salud = 100
else [vidas vidas < 0]
    sub -> sub : 4b: salud = 0
end
deactivate sub

== Ganar una vida extra ==

juego -> juego : 5: sincronizarSubmarino()
activate juego
juego -> juego : 6: vidasPorPuntos = puntos / 500
opt [vidasPorPuntos vidasPorPuntos > vidasExtra]
    juego -> sub : 7: sumarVidas(vidasPorPuntos - vidasExtra)
    activate sub
    sub -> sub : 8: vidas += vidasExtra
    sub --> juego
    deactivate sub
    juego -> juego : 9: vidasExtra = vidasPorPuntos
end
deactivate juego
@enduml
