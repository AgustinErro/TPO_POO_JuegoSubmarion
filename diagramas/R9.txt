@startuml
title R9 - Incremento de Dificultad por Nivel

participant "juego:JuegoSubmarino" as juego
participant "barco:Barco" as barco
participant "carga:CargaProfundidad" as carga

== Al pasar de nivel ==

juego -> juego : 1: pasarNivel()
activate juego


loop [3 3 veces]
    juego -> barco : 3: <<create>> Barco()
    activate barco
    barco --> juego
    deactivate barco
end

deactivate juego

== Al disparar una carga ==

activate barco
barco -> barco : 4: dispararCarga()
activate barco
barco -> carga : 5: <<create>> CargaProfundidad()
activate carga
carga --> barco
deactivate carga
deactivate barco
@enduml
