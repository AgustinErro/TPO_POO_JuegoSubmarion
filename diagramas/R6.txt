@startuml
title R6 - Cálculo de Daño y Puntaje

participant "juego:JuegoSubmarino" as juego
participant "barco:Barco" as barco
participant "carga:CargaProfundidad" as carga
participant "sub:Submarino" as sub

juego -> juego : 1: sincronizarSubmarino()
activate juego

loop [por cada Barco]
    juego -> barco : 2: hayCargasExplotadas()
    activate barco
    loop [por cada CargaProfundidad]
        barco -> carga : 3: isExploto()
        activate carga
        carga --> barco : true
        deactivate carga
        barco -> carga : 4: getCentro()
        activate carga
        carga --> barco : [centroX, centroY]
        deactivate carga
        barco -> barco : 5: cargasDisparadas.remove(i)
    end
    barco --> juego : centrosExplocion[]
    deactivate barco
end

juego -> sub : 6: detectarDano(centrosExplocion)
activate sub

loop [por cada explosión]
    sub -> sub : 7: distanciaAlBorde
    sub -> sub : 8: calcularDano(distancia)
    opt [salud salud <= 0]
        sub -> sub : 9: perderVida()
    end
end

sub --> juego : puntosGanados
deactivate sub
juego -> juego : 10: puntos += puntosGanados
deactivate juego
@enduml
