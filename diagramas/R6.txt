@startuml
title R6 - Cálculo de Daño y Puntaje

participant JuegoSubmarino
participant Barco
participant CargaProfundidad
participant Submarino

JuegoSubmarino -> JuegoSubmarino : sincronizarSubmarino()

loop por cada Barco
    JuegoSubmarino -> Barco : hayCargasExplotadas()
    loop por cada CargaProfundidad
        Barco -> CargaProfundidad : isExploto()
        CargaProfundidad --> Barco : true
        Barco -> CargaProfundidad : getCentro()
        CargaProfundidad --> Barco : [centroX, centroY]
        Barco -> Barco : cargasDisparadas.remove(i)
    end
    Barco --> JuegoSubmarino : centrosExplocion[]
end

JuegoSubmarino -> Submarino : detectarDano(centrosExplocion)

loop por cada explosión
    Submarino -> Submarino : distanciaAlBorde = sqrt(dx² + dy²)
    alt distancia > 100
        note right : +30 puntos, sin daño
    else distancia 51 a 100
        note right : +10 puntos, -30 salud
    else distancia 11 a 50
        note right : -50 salud
    else distancia <= 10
        note right : -100 salud
    end
    alt salud <= 0
        Submarino -> Submarino : perderVida()
    end
end

Submarino --> JuegoSubmarino : puntosGanados
JuegoSubmarino -> JuegoSubmarino : puntos += puntosGanados
@enduml
