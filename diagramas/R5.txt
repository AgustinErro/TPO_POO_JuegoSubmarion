@startuml
title R5 - Detonación a Profundidad Aleatoria

participant "juego:JuegoSubmarino" as juego
participant "barco:Barco" as barco
participant "carga:CargaProfundidad" as carga
participant "area:Area" as area

juego -> barco : 1: moverCargas()
activate barco

loop [por cada CargaProfundidad]
    barco -> carga : 2: moverAbajo()
    activate carga

    alt [posY < profExplosion]
        carga -> area : 3: estaDentroVertical(nuevaY, alto)
        activate area
        area --> carga : true
        deactivate area
        carga -> carga : 4: posY += velocidad
    else [posY >= profExplosion]
        carga -> carga : 5: exploto = true
    end

    carga --> barco
    deactivate carga
end

barco --> juego
deactivate barco
@enduml
