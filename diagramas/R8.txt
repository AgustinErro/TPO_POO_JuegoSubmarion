@startuml
title R8 - Cambio de Nivel

participant "controlador:Controlador" as controlador
participant "juego:JuegoSubmarino" as juego
participant "barco:Barco" as barco

controlador -> juego : 1: sincronizarNivel()
activate juego

loop [por cada Barco]
    juego -> barco : 2: isInactivo()
    activate barco
    barco --> juego : true
    deactivate barco
    juego -> juego : 3: barcos.remove(i)
end

opt [barcos.isEmpty()]
    juego -> juego : 4: pasarNivel()
    juego -> juego : 5: nivel += 1
    juego -> juego : 6: puntos += 200
    loop [3 veces]
        juego -> barco : <<create>>\n7: Barco(areaJuego, 90, 30, velocidadBarco, nivel)
        activate barco
        barco --> juego
        deactivate barco
        juego -> juego : 8: barcos.add(barco)
    end
end

juego --> controlador
deactivate juego
@enduml
